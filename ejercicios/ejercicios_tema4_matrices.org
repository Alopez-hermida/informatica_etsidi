* Multiplicación de matrices

Escribe un programa que realice la multiplicación matricial entre las siguientes matrices: 

#+begin_export latex
\[
P = \left[ 
\begin{tabular}{rrr}
1 & 3 & -4\\
1 & 1 & -2\\
-1 & -2 & 5\\
\end{tabular}
\right]
\]
\[
Q = \left[
\begin{tabular}{rrr}
8 & 3 & 0\\
3 & 10 & 2\\
0 & 2 & 6\\
\end{tabular}
\right]
\]
#+end_export


#+BEGIN_SRC C
#include <stdio.h>

int main()
{
  int p[3][3] = {{1, 3, -4}, {1, 1, -2}, {-1, -2, 5}};
  int q[3][3] = {{8, 3, 0}, {3, 10, 2}, {0, 2, 6}};
  int r[3][3] = {0}; // Matriz multiplicacion

  int i, j, k; //Indices de las matrices
  
  //Multiplica las matrices p y q
  for(i = 0; i < 3; i++)
    {
      for(j = 0; j < 3; j++)
	{
	  for(k = 0; k < 3; k++)
	    r[i][j] += p[i][k] * q[k][j];
	}
    }
  // Imprime el resultado
  for(i = 0; i < 3; i++)
    {
      for(j = 0; j < 3; j++)
	    printf("%d\t", r[i][j]);
      printf("\n"); // Un salto de linea tras cada fila
    }
}
#+END_SRC

|  11 |   9 | -10 |
| -14 | -13 |  26 |
